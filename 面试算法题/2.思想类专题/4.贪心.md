## 455. 分发饼干

```python
class Solution:
    def findContentChildren(self, g: List[int], s: List[int]) -> int:
        if not s:
            return 0
        g.sort()
        s.sort()
        i, j = 0, 0
        # 此处直接思路是要做双层循环，但是使用一个 and 简化了双层循环
        while i < len(g) and j < len(s):
            if s[j] >= g[i]:
                i += 1
                j += 1
            else:
                j += 1
        return i

```

## 122. 买卖股票的最佳时机 II

```PYTHON
class Solution:
    def maxProfit(self, prices: List[int]) -> int:
        res = 0
        for i in range(1, len(prices)):
            res += max(0, prices[i] - prices[i-1])
        return res
```

## 55. 跳跃游戏

```PYTHON
class Solution:
    def canJump(self, nums: List[int]) -> bool:
        step, i = 1, 0
        while i < len(nums) and step:
            step -= 1
            step = max(step, nums[i])
            i += 1
        return i == len(nums)
```

## 1005. K 次取反后最大化的数组和

```PYTHON
class Solution:
    def largestSumAfterKNegations(self, nums: List[int], k: int) -> int:
        nums.sort()
        for i in range(len(nums)):
            if nums[i] < 0 and k:
                k -= 1
                nums[i] = -nums[i]
            else:
                break
        if not k:
            return sum(nums)
        nums.sort()
        if nums[0] == 0:
            return sum(nums)
        if k % 2 == 1:
            return sum(nums) - 2 * nums[0]
        else:
            return sum(nums)

```

## 860. 柠檬水找零

```PYTHON
class Solution:
    def lemonadeChange(self, bills: List[int]) -> bool:
        ht = {5: 0, 10: 0, 20: 0}
        for i in range(len(bills)):
            ht[bills[i]] += 1
            if bills[i] == 10:
                if ht[5]:
                    ht[5] -= 1
                else:
                    return False
            elif bills[i] == 20:
                if ht[5] and ht[10]:
                    ht[5] -= 1
                    ht[10] -= 1
                elif ht[5] > 2:
                    ht[5] -= 3
                else:
                    return False
        return True
```
